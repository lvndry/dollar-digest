# Search Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a subscriber-only `/search` page where users can search the full article archive using structured filters (date, topic, tags) and free-text, with a typed query bar and visual filter panels that stay in sync via URL params.

**Architecture:** Server Component page reads URL params, verifies archive access with the existing `canAccessArchive()` guard, calls a new `searchArticles()` function in `src/lib/search.ts` that builds dynamic Drizzle queries, then renders results using `ArticleCard` in a grid with link-based pagination. A `SearchFilters` Client Component owns the query bar and filter panels and updates the URL on change.

**Tech Stack:** Next.js 15 App Router, Drizzle ORM + libsql (Turso), Bun test, Tailwind v4, `next/navigation` for client-side URL updates.

import { and, count, desc, eq, like, or } from "drizzle-orm";
import { db } from "./db";
import { articles } from "./schema";
import type { Article } from "./schema";

export type SearchFilters = {
  text?: string;
  date?: string;
  topic?: "tech" | "politics";
  tags?: string[];
  page: number;
};

export type SearchResult = {
  articles: Article[];
  total: number;
  pages: number;
};

const PAGE_SIZE = 20;

function normalizeDate(raw: string | null): string | undefined {
  if (!raw) return undefined;
  if (/^\d{4}-\d{2}-\d{2}$/.test(raw)) return raw;
  const match = raw.match(/^(\d{2})-(\d{2})-(\d{4})$/);
  if (match) return `${match[3]}-${match[1]}-${match[2]}`;
  return undefined;
}

export function parseSearchParams(params: URLSearchParams): SearchFilters {
  const topic = params.get("topic");
  const tagsRaw = params.get("tags");
  const text = params.get("text") ?? undefined;
  const page = Math.max(1, parseInt(params.get("page") ?? "1", 10));

  return {
    topic: topic === "tech" || topic === "politics" ? topic : undefined,
    tags: tagsRaw
      ? tagsRaw
          .split(",")
          .map((tag) => tag.trim())
          .filter(Boolean)
      : undefined,
    date: normalizeDate(params.get("date")),
    text: text || undefined,
    page,
  };
}

export function parseQueryBar(raw: string): URLSearchParams {
  const params = new URLSearchParams();
  let remaining = raw.trim();

  const topicMatch = remaining.match(/\btopic:(tech|politics)\b/i);
  if (topicMatch) {
    params.set("topic", topicMatch[1].toLowerCase());
    remaining = remaining.replace(topicMatch[0], "").trim();
  }

  const dateMatch = remaining.match(/\bdate:(\S+)/i);
  if (dateMatch) {
    const normalized = normalizeDate(dateMatch[1]);
    if (normalized) params.set("date", normalized);
    remaining = remaining.replace(dateMatch[0], "").trim();
  }

  // tags: everything after "tags:" up to the next "word:" token or end
  const tagsMatch =
    remaining.match(/\btags:\s*(.+?)(?=\s+\w+:|$)/i) ??
    remaining.match(/\btags:\s*(.+)/i);
  if (tagsMatch) {
    const tagValues = tagsMatch[1]
      .split(/,|\s+or\s+/i)
      .map((tag) => tag.trim())
      .filter(Boolean);
    if (tagValues.length > 0) params.set("tags", tagValues.join(","));
    remaining = remaining.replace(tagsMatch[0], "").trim();
  }

  const freeText = remaining.replace(/\s+/g, " ").trim();
  if (freeText) params.set("text", freeText);

  return params;
}

export function reconstructQuery(params: URLSearchParams): string {
  const parts: string[] = [];

  const topic = params.get("topic");
  if (topic) parts.push(`topic:${topic}`);

  const tags = params.get("tags");
  if (tags) parts.push(`tags:${tags}`);

  const date = params.get("date");
  if (date) {
    const match = date.match(/^(\d{4})-(\d{2})-(\d{2})$/);
    parts.push(`date:${match ? `${match[2]}-${match[3]}-${match[1]}` : date}`);
  }

  const text = params.get("text");
  if (text) parts.push(text);

  return parts.join(" ");
}

export async function searchArticles(filters: SearchFilters): Promise<SearchResult> {
  const conditions = [];

  if (filters.date) {
    conditions.push(eq(articles.digestDate, filters.date));
  }
  if (filters.topic) {
    conditions.push(eq(articles.category, filters.topic));
  }
  if (filters.tags && filters.tags.length > 0) {
    conditions.push(or(...filters.tags.map((tag) => like(articles.tags, `%"${tag}"%`)))!);
  }
  if (filters.text) {
    const term = `%${filters.text}%`;
    conditions.push(or(like(articles.title, term), like(articles.summary, term))!);
  }

  const where = conditions.length > 0 ? and(...conditions) : undefined;
  const offset = (filters.page - 1) * PAGE_SIZE;

  const [rows, [countRow]] = await Promise.all([
    db
      .select()
      .from(articles)
      .where(where)
      .orderBy(desc(articles.importanceScore))
      .limit(PAGE_SIZE)
      .offset(offset),
    db.select({ count: count() }).from(articles).where(where),
  ]);

  const total = countRow?.count ?? 0;

  return {
    articles: rows,
    total,
    pages: Math.ceil(total / PAGE_SIZE),
  };
}

interface SearchPaginationProps {
  currentPage: number;
  totalPages: number;
  searchParams: Record<string, string>;
}

function buildPageUrl(page: number, searchParams: Record<string, string>): string {
  const params = new URLSearchParams({ ...searchParams, page: String(page) });
  return `/search?${params.toString()}`;
}

export function SearchPagination({
  currentPage,
  totalPages,
  searchParams,
}: SearchPaginationProps) {
  if (totalPages <= 1) return null;

  return (
    <nav
      className="flex items-center justify-center gap-2 mt-12"
      aria-label="Search results pages"
    >
      {currentPage > 1 && (
        <a
          href={buildPageUrl(currentPage - 1, searchParams)}
          className="font-ui text-[0.6rem] tracking-widest uppercase px-3 py-1.5 transition-opacity duration-150 hover:opacity-60"
          style={{ color: "var(--ink-muted)" }}
        >
          ← Prev
        </a>
      )}

      {Array.from({ length: totalPages }, (_, index) => index + 1).map((page) => {
        const active = page === currentPage;
        return (
          <a
            key={page}
            href={buildPageUrl(page, searchParams)}
            className="font-ui text-[0.6rem] tracking-widest uppercase px-3 py-1.5 transition-opacity duration-150"
            style={{
              color: active ? "var(--ink)" : "var(--ink-muted)",
              fontWeight: active ? "500" : "400",
              opacity: active ? 1 : undefined,
            }}
            aria-current={active ? "page" : undefined}
          >
            {page}
          </a>
        );
      })}

      {currentPage < totalPages && (
        <a
          href={buildPageUrl(currentPage + 1, searchParams)}
          className="font-ui text-[0.6rem] tracking-widest uppercase px-3 py-1.5 transition-opacity duration-150 hover:opacity-60"
          style={{ color: "var(--ink-muted)" }}
        >
          Next →
        </a>
      )}
    </nav>
  );
}

"use client";

import { useRouter, useSearchParams } from "next/navigation";
import { useCallback, useEffect, useState } from "react";
import { parseQueryBar, reconstructQuery } from "@/lib/search";

export function SearchFilters() {
  const router = useRouter();
  const searchParams = useSearchParams();

  const [queryBarValue, setQueryBarValue] = useState(() =>
    reconstructQuery(searchParams),
  );

  useEffect(() => {
    setQueryBarValue(reconstructQuery(searchParams));
  }, [searchParams]);

  const navigate = useCallback(
    (params: URLSearchParams) => {
      params.delete("page");
      router.push(`/search?${params.toString()}`);
    },
    [router],
  );

  function handleQueryBarSubmit(event: React.FormEvent) {
    event.preventDefault();
    navigate(parseQueryBar(queryBarValue));
  }

  function handleTopicChange(value: string) {
    const params = new URLSearchParams(searchParams.toString());
    if (value) {
      params.set("topic", value);
    } else {
      params.delete("topic");
    }
    navigate(params);
  }

  function handleTagsChange(value: string) {
    const params = new URLSearchParams(searchParams.toString());
    const tags = value
      .split(",")
      .map((tag) => tag.trim())
      .filter(Boolean)
      .join(",");
    if (tags) {
      params.set("tags", tags);
    } else {
      params.delete("tags");
    }
    navigate(params);
  }

  function handleDateChange(value: string) {
    const params = new URLSearchParams(searchParams.toString());
    if (value) {
      params.set("date", value);
    } else {
      params.delete("date");
    }
    navigate(params);
  }

  const currentTopic = searchParams.get("topic") ?? "";
  const currentTags = searchParams.get("tags") ?? "";
  const currentDate = searchParams.get("date") ?? "";

  return (
    <div className="mb-10 space-y-5">
      {/* Query bar */}
      <form onSubmit={handleQueryBarSubmit} className="flex gap-2">
        <input
          type="text"
          value={queryBarValue}
          onChange={(event) => setQueryBarValue(event.target.value)}
          placeholder="topic:politics tags:AI,VC date:04-10-2026 Fed"
          className="flex-1 font-ui text-[0.75rem] px-4 py-2.5 bg-transparent border transition-colors duration-150 focus:outline-none"
          style={{
            color: "var(--ink)",
            borderColor: "var(--border)",
          }}
        />
        <button
          type="submit"
          className="font-ui text-[0.6rem] tracking-widest uppercase px-4 py-2.5 transition-opacity duration-150 hover:opacity-75 shrink-0"
          style={{ color: "var(--bg)", backgroundColor: "var(--ink)" }}

          Search
        </button>
      </form>

      {/* Visual filter panels */}
      <div className="flex flex-wrap gap-3">
        {/* Topic */}
        <select
          value={currentTopic}
          onChange={(event) => handleTopicChange(event.target.value)}
          className="font-ui text-[0.6rem] tracking-widest uppercase px-3 py-2 border bg-transparent cursor-pointer transition-colors duration-150 focus:outline-none"

        <input
          type="date"

```tsx
import { redirect } from "next/navigation";
import { Suspense } from "react";
import { auth } from "@/auth";
import { canAccessArchive } from "@/lib/access";
import { parseSearchParams, searchArticles } from "@/lib/search";
import { SearchFilters } from "@/components/SearchFilters";
import { SearchPagination } from "@/components/SearchPagination";
import { ArticleCard } from "@/components/ArticleCard";

interface SearchPageProps {
  searchParams: Promise<Record<string, string | string[] | undefined>>;
}

export const metadata = { title: "Search — The One Dollar Digest" };

export default async function SearchPage({ searchParams }: SearchPageProps) {
  const session = await auth();

  if (!canAccessArchive(session)) {
    redirect("/login?callbackUrl=/search");
  }

  const rawParams = await searchParams;

  // Flatten string | string[] | undefined → string, drop arrays
  const flatParams = Object.fromEntries(
    Object.entries(rawParams).filter(
      (entry): entry is [string, string] => typeof entry[1] === "string",
    ),
  );

  const urlParams = new URLSearchParams(flatParams);
  const filters = parseSearchParams(urlParams);
  const hasFilters = !!(filters.text ?? filters.topic ?? filters.tags ?? filters.date);

  const {
    articles: results,
    total,
    pages,
  } = hasFilters ? await searchArticles(filters) : { articles: [], total: 0, pages: 0 };

  return (
    <main className="max-w-5xl mx-auto px-6 py-12">
      <h1
        className="font-display italic mb-10 fade-in"
        style={{
          color: "var(--ink)",
          fontSize: "clamp(2rem, 5vw, 3rem)",
          letterSpacing: "-0.025em",
          lineHeight: "1",
        }}
      >
        Search
      </h1>

      <Suspense>
        <SearchFilters />
      </Suspense>

      {hasFilters && (
        <>
          <div
            className="mb-6 font-ui text-[0.6rem] tracking-widest uppercase"
            style={{ color: "var(--ink-muted)" }}
          >
            {total === 0 ? "No results" : `${total} result${total === 1 ? "" : "s"}`}
          </div>

          {results.length > 0 && (
            <>
              <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-x-7 gap-y-10 [&>*]:min-w-0">
                {results.map((article, index) => (
                  <ArticleCard key={article.id} article={article} index={index} />
                ))}
              </div>

              <SearchPagination
                currentPage={filters.page}
                totalPages={pages}
                searchParams={flatParams}
              />
            </>
          )}
        </>
      )}

      {!hasFilters && (
        <p className="font-ui text-[0.75rem]" style={{ color: "var(--ink-muted)" }}>
          Use the filters above to search the archive.
        </p>
      )}
    </main>
  );
}
```

## Self-review notes

- `searchArticles()` only runs when at least one filter is set — no full-table scan on page load.
- `canAccessArchive` is called server-side in the page; the Client Component never receives raw session data.
- `SearchFilters` uses `useSearchParams()` directly so it always reflects the current URL without needing props drilled from the Server Component.
- Tags matching uses `%"TAG"%` with quoted delimiters, which correctly distinguishes `"AI"` from `"MAIN"` in a JSON array like `["AI","startups"]`.
- Pagination is purely link-based — no JS needed to navigate pages.
